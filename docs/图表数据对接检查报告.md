**文件**：`StatisticsServiceImpl.java`

**添加**：
```java
@Override
public List<Map<String, Object>> getContractMonthlyStats() {
    List<Map<String, Object>> monthlyStats = new ArrayList<>();
    
    // 近 6 个月每月的合同金额
    for (int i = 5; i >= 0; i--) {
        LocalDate firstDay = LocalDate.now().minusMonths(i).withDayOfMonth(1);
        LocalDate lastDay = firstDay.plusMonths(1).minusDays(1);
        
        LambdaQueryWrapper<Contract> wrapper = new LambdaQueryWrapper<>();
        wrapper.between(Contract::getSignDate, firstDay, lastDay);
        
        // 使用 MP 查询总和
        List<Map<String, Object>> result = contractMapper.selectMaps(
            new LambdaQueryWrapper<Contract>()
                .select(Contract::getAmount)
                .between(Contract::getSignDate, firstDay, lastDay)
        );
        
        BigDecimal amount = result.stream()
            .map(m -> (BigDecimal)m.get("amount"))
            .filter(Objects::nonNull)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        Map<String, Object> monthStat = new HashMap<>();
        monthStat.put("month", firstDay.format(DateTimeFormatter.ofPattern("yyyy-MM")));
        monthStat.put("amount", amount);
        monthlyStats.add(monthStat);
    }
    
    return monthlyStats;
}
```

// 2. initCustomerGrowthChart 函数
const initCustomerGrowthChart = async () => {
  if (!customerGrowthChart.value) return
  
  try {
    const res = await getCustomerGrowthStats()
    // ✅ 使用 dailyStats 数组
    const growthData = res.data.dailyStats || []
    
    growthChartInstance = echarts.init(customerGrowthChart.value)
    
    const option = {
      tooltip: { trigger: 'axis' },
      xAxis: {
        type: 'category',
        data: growthData.map(item => item.date)  // ✅ 使用 date 字段
      },
      yAxis: { type: 'value' },
      series: [{
        data: growthData.map(item => item.count),  // ✅ 使用 count 字段
        type: 'line',
        smooth: true,
        itemStyle: { color: '#409EFF' },
        areaStyle: {

            { offset: 1, color: 'rgba(64, 158, 255, 0.05)' }

        }
      }]
