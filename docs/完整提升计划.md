#### Day 7：数据库索引优化（1 小时）

**创建文件**：`docs/sql/indexes.sql`

```sql
-- 客户表（4 个索引）
CREATE INDEX idx_customer_level ON crm_customer(level);
CREATE INDEX idx_customer_status ON crm_customer(status);
CREATE INDEX idx_customer_owner ON crm_customer(owner_id);
CREATE INDEX idx_customer_create ON crm_customer(create_time);

-- 跟进表（2 个索引）
CREATE INDEX idx_follow_customer ON crm_follow_up(customer_id);
CREATE INDEX idx_follow_create ON crm_follow_up(create_time);

-- 商机表（3 个索引）
CREATE INDEX idx_opportunity_customer ON crm_opportunity(customer_id);
CREATE INDEX idx_opportunity_stage ON crm_opportunity(stage);
CREATE INDEX idx_opportunity_create ON crm_opportunity(create_time);

-- 合同表（3 个索引）
CREATE INDEX idx_contract_customer ON crm_contract(customer_id);
CREATE INDEX idx_contract_status ON crm_contract(status);
CREATE INDEX idx_contract_create ON crm_contract(create_time);
```

**效果**：查询速度提升 50%+

#### Day 8：全局异常处理增强（2 小时）

```java
public enum ErrorCode {
    // 认证错误
    UNAUTHORIZED(401, "未授权"),
    TOKEN_EXPIRED(401, "Token 已过期"),
    
    // 权限错误
    FORBIDDEN(403, "权限不足"),
    
    // 资源错误
    NOT_FOUND(404, "资源不存在"),
    CUSTOMER_NOT_FOUND(404, "客户不存在"),
    FOLLOW_UP_NOT_FOUND(404, "跟进记录不存在"),
    OPPORTUNITY_NOT_FOUND(404, "商机不存在"),
    CONTRACT_NOT_FOUND(404, "合同不存在"),
    
    // 业务错误
    DATA_EXISTS(500, "数据已存在"),
    OPERATION_FAILED(500, "操作失败");
}
```
